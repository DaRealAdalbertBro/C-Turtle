199  AffineTransform& rotate(float theta) {
200  //6.28319 is a full circle in radians.
201  if(rotation + theta > 6.28319){//Loop back around. Avoid overflow.
202  theta = (rotation + theta) - 6.28319;
203  setRotation(0);
204  }
205  const float c = std::cos(theta);
206  const float s = std::sin(theta);
207 
208  const float new00 = at(0, 0) * c + at(0, 1) * s;
209  const float new01 = at(0, 0) * -s + at(0, 1) * c;
210  const float new10 = at(1, 0) * c + at(1, 1) * s;
211  const float new11 = at(1, 0) * -s + at(1, 1) * c;
212 
213  at(0, 0) = new00;
214  at(0, 1) = new01;
215  at(1, 0) = new10;
216  at(1, 1) = new11;
217 
218  rotation += theta;
219 
220  return *this;
221  }

276  AffineTransform& concatenate(const AffineTransform& t){
277  const float new00 = constAt(0,0) * t.constAt(0,0) + constAt(0,1) * t.constAt(1,0);
278  const float new01 = constAt(0,0) * t.constAt(0,1) + constAt(0,1) * t.constAt(1,1);

333  Point transform(Point in, Point* dst = nullptr) const{
334  Point temp;
335  Point* dstPtr = (dst == nullptr) ? &temp : dst;
336  dstPtr->x = (constAt(0, 0) * (in.x) + constAt(0, 1) * (in.y) + constAt(0, 2));
337  dstPtr->y = (constAt(1, 0) * (in.x) + constAt(1, 1) * (in.y) + constAt(1, 2));
338  return *dstPtr;

571  void drawLine(Image& imgRef, int x1, int y1, int x2, int y2, Color c, unsigned int width){
572  if (x1 == x2 && y1 == y2) {
573  return;
574  }
575  const double xoffs = std::abs(x1 - x2);
576  const double yoffs = std::abs(y1 - y2);
577  const double woffs = width / 2.0;
578  const int xadjacent = yoffs * woffs / std::sqrt(std::pow(xoffs, 2) + std::pow(yoffs, 2));
579  const int yadjacent = xoffs * woffs / std::sqrt(std::pow(xoffs, 2) + std::pow(yoffs, 2));
580 
581  //Clockwise
582  cimg_library::CImg<int> points(4, 2);
583  points(0, 0) = x1 - xadjacent;
584  points(0, 1) = y1 + yadjacent;
585  points(1, 0) = x1 + xadjacent;
586  points(1, 1) = y1 - yadjacent;
587  points(2, 0) = x2 + xadjacent;
588  points(2, 1) = y2 - yadjacent;
589  points(3, 0) = x2 - xadjacent;
590  points(3, 1) = y2 + yadjacent;
591  imgRef.draw_polygon(points, c.rgbPtr());
592  }

623  void Polygon::draw(const AffineTransform& t, Image& imgRef, Color c) {
624  if (points.empty())
625  return;
626  /*CImg is a pain in the butt and requires all polygons to be
627  passed in as an instance of the image object. Therefore,
628  we can specify an "int" image with a width of 2 (x,y) and height
629  of the total number of elements in the point vector.*/
630  cimg::CImg<int> passPts(points.size(), 2);
631 
632  for (int i = 0; i < points.size(); i++) {
633  const Point pt = t(points[i]);
634  passPts(i, 0) = pt.x;
635  passPts(i, 1) = pt.y;
636  }
637 
638  imgRef.draw_polygon(passPts, c.rgbPtr());
639  }

641  void Polygon::drawOutline(const AffineTransform& t, Image& imgRef, Color c){
642  if(points.size() < 2)
643  return;
644 
645  typedef decltype(points.begin()) iter_t;
646 
647  iter_t cur = points.begin()++;
648  iter_t last = points.begin();
649 
650  while(cur != points.end()){
651  Point a = t(*cur);
652  Point b = t(*last);
653  imgRef.draw_line(a.x, a.y, b.x, b.y, c.rgbPtr());
654  last = cur;
655  cur++;
656  }
657 
658  /*Draw a line between the first and last points.*/
659  Point a = t(points.front());
660  Point b = t(points.back());
661  imgRef.draw_line(a.x, a.y, b.x, b.y, c.rgbPtr());
662  }
