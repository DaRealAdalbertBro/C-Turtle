143  void TurtleScreen::update(bool doRedraw, bool input) {
144  /*Resize canvas.*/
145  if(display.is_resized()){
146  display.resize();
147  redraw();
148  }else if ((doRedraw && delayMS > -1))
149  redraw();
150 
152  if(cachedEvents.empty() || !input)
153  return;//No events to process.
154 
155  if(!cacheMutex.try_lock())
156  return;
157 
158  for (InputEvent& event : cachedEvents) {
159  if (event.type) {//process keyboard event
160  KeyFunc& keyFunc = *reinterpret_cast<KeyFunc*> (event.cbPointer);
161  keyFunc();
162  } else {//process mouse event
163  MouseFunc& mFunc = *reinterpret_cast<MouseFunc*> (event.cbPointer);
164  mFunc(event.mX, event.mY);
165  }
166  }
167 
168  cachedEvents.clear();
169  cacheMutex.unlock();
170  }

191  void TurtleScreen::redraw() {
192  if (getIsClosed() || delayMS <= -1)
193  return;
194 
195  //Moved to initialization of the event thread.
196  // update();//TODO: Move this to somewhere more appropriate.
197  AffineTransform screen = screentransform();
198 
199  for (SceneObject& object : objects) {
200  AffineTransform t(screen.copyConcatenate(object.transform));
201  Color& color = object.color;
202  IDrawableGeometry* geom = object.geom.get();
203  if (!object.text.empty()) {
204  Point trans = t.getTranslation();
205  canvas.draw_text(trans.x, trans.y, object.text.c_str(), object.color.rgbPtr());
206  } else if (object.stamp && object.unownedGeom != nullptr) {
207  object.unownedGeom->draw(screen.copyConcatenate(object.transform), canvas, object.color);
208  } else geom->draw(t, canvas, color);
209  }
210 
211  for (RawTurtle* turtle : turtles) {
212  turtle->draw(screen, canvas);
213  }
214  sleep(delayMS);
215  swap();
216  }

218  void TurtleScreen::initEventThread() {
219  eventThread.reset(new std::thread([&]() {
220  bool mButtons[3] = {false, false, false};
221  while (!display.is_closed() && !killEventThread) {
222  //Updates all input.
223 
224  display.paint();
225  if(!cacheMutex.try_lock()){
226  std::this_thread::yield();
227  continue;
228  }
229 
230  AffineTransform mouseOffset = screentransform();
231 
232  //Update mouse button input.
233  const unsigned int button = display.button();
234  bool buttons[3] = {
235  button & 1, //left
236  button & 2, //right
237  button & 4//middle
238  };
239 
240  for (int i = 0; i < 3; i++) {
241  if (!(!mButtons[i] && buttons[i]))//is this button state "down"?
242  continue; //if not, skip its processing loop.
243 
244  for (MouseFunc& func : mouseBindings[i]) {
245  //append to the event cache.
246  InputEvent e;
247  e.type = false;
248  e.mX = display.mouse_x() - mouseOffset.getTranslateX();
249  e.mY = display.mouse_y() - mouseOffset.getTranslateY();
250  e.cbPointer = reinterpret_cast<void*> (&func);
251  cachedEvents.push_back(e);
252  }
253  }
254 
255 
256  if(display.is_key()){
257  //Update keyboard key input.
258  for (auto& kbPair : keyBindings) {
259  KeyboardKey key = kbPair.first;
260  //is key for assigned pair being called?
261  //if not, skip its processing loop.
262  if (!display.is_key((unsigned int) key))
263  continue;
264  auto& callbackList = kbPair.second;
265  for (KeyFunc& func : callbackList) {
266  InputEvent e;
267  e.type = true;
268  e.cbPointer = reinterpret_cast<void*> (&func);
269  cachedEvents.push_back(e);
270  }
271  }
272  }
273 
274  mButtons[0] = buttons[0];
275  mButtons[1] = buttons[1];
276  mButtons[2] = buttons[2];
277 
278  cacheMutex.unlock();
279  }
280  }));
281  }

324  void RawTurtle::clearstamp(int stampid) {
325  auto iter = objects.begin();//iterator which holds an iterator to the screen's scene list.
326 
327  while (iter != objects.end()) {
328  auto& objIter = *iter;
329  if (objIter->stamp && objIter->stampid == stampid) {
330  break;
331  }
332  iter++;
333  }
334 
335  if (iter != objects.end()){
336  objects.erase(iter);
337 
338  if(screen != nullptr){
339  screen->getScene().erase(*iter);
340  }
341  }

344  void RawTurtle::clearstamps(int stampid) {
345  typedef decltype(objects.begin()) iter_t;
346 
347  std::list<iter_t> removals;
348 
349  iter_t iter = objects.begin();
350  while (iter != objects.end()) {
351  auto& objIter = *iter;
352  if (stampid < 0 ? objIter->stamp : (objIter->stamp && objIter->stampid <= stampid)) {
353  removals.push_back(iter);
354  }
355  iter++;
356  }
357 
358  for (iter_t& iter : removals) {
359  screen->getScene().erase(*iter);
360  objects.erase(iter);
361  }
362  }

512  void RawTurtle::travelTo(const AffineTransform& dest) {
513  AffineTransform begin;
514  begin.assign(transform);
515 
516  if (screen != nullptr ? !screen->getIsClosed() : false) {//no point in animating with no screen
517  const float duration = getAnimMS();
518  const unsigned long startTime = epochTime();
519  const unsigned long endTime = duration + startTime;
520 
521  AffineTransform start;
522  start.assign(transform);
523 
524  float progress = duration == 0 ? 1 : 0;
525  unsigned long lastTime = startTime;
526 
527  while (progress < 1.0f) {
528  //We use the time between animation frames to smooth out
529  //our animations, making them take the same amount of time
530  //regardless of how it's performance.
531  unsigned long curTime = epochTime();
532  unsigned long delta = curTime - lastTime;
533 
534  if(tracing){
535  //Push up a temporary line if we're tracing
536  //This is to draw a pen line from where we were to where we
537  //are at in the animation.
538  transform.assign(start.lerp(dest, progress));
539  traceLines.push_back(std::make_pair(penColor, Line(transform.getTranslation(), begin.getTranslation(), penWidth)));
540  updateParent(true, false);
541  traceLines.pop_back();
542  }else{
543  transform.assign(start.lerp(dest, progress));
544  updateParent(true, false);
545  }
546 
547  progress += ((curTime - startTime) / duration) * (delta / 1000.0);
548  lastTime = curTime;
549  }
550  }
551 
552  //Contents of PushCurrent moved here because every function
553  //that called this one called PushCurrent immediately after it.
554  if(tracing){
555  traceLines.push_back(std::make_pair(penColor, Line(begin.getTranslation(), dest.getTranslation(), penWidth)));
556  }
557 
558  transform.assign(dest);
559 
560  if (filling)
561  fillAccum.points.push_back(transform.getTranslation());
562  transformStack.push_back(transform);
563  transform = transformStack.back();
564 
565  updateParent();
566  }
