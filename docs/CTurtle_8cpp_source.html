146  void TurtleScreen::update(bool invalidateDraw, bool input) {
147  /*Resize canvas.*/
148  if(display.is_resized()){
149  display.resize();
150  invalidateDraw = true;
151  }
152  redraw(invalidateDraw);
153 
154  if(input && !timerBindings.empty()){
155  //Call timer bindings first.
156  uint64_t curTime = epochTime();
157  for(auto& timer : timerBindings){
158  auto& func = std::get<0>(timer);
159  uint64_t reqTime = std::get<1>(timer);
160  uint64_t& lastCalled = std::get<2>(timer);
161 
162  if(curTime >= lastCalled + reqTime){
163  lastCalled = curTime;
164  func();
165  }
166  }
167  }
168 
170  if(cachedEvents.empty() || !input)
171  return;//No events to process.
172 
173  cacheMutex.lock();
174 
175  for (InputEvent& event : cachedEvents) {
176  if (event.type) {//process keyboard event
177  KeyFunc& keyFunc = *reinterpret_cast<KeyFunc*> (event.cbPointer);
178  keyFunc();
179  } else {//process mouse event
180  MouseFunc& mFunc = *reinterpret_cast<MouseFunc*> (event.cbPointer);
181  mFunc(event.mX, event.mY);
182  }
183  }
184 
185  cachedEvents.clear();
186  cacheMutex.unlock();
187  }

211  void TurtleScreen::redraw(bool invalidate) {
212  if(getIsClosed())
213  return;
214  int fromBack = 0;
215  bool hasInvalidated = invalidate;
216 
217  //Handle resizes.
218  if (display.window_width() != canvas.width() || display.window_height() != canvas.height()) {
219  canvas.resize(display);
220  hasInvalidated = true;
221  }
222 
223  if (lastTotalObjects != objects.size()) {
224  if (lastTotalObjects > objects.size()) {
225  //invalidate; size of objects has changed..
226  //it can check this case automatically.
227  //all other cases which cause invalidation
228  //must call this function with the first param
229  //set to true to force invalidation.
230  hasInvalidated = true;
231  } else {
232  fromBack = objects.size() - lastTotalObjects;
233  }
234  };
235 
236  if(hasInvalidated){
237  if (!backgroundImage.is_empty()) {
238  canvas.assign(backgroundImage);
239  } else {
240  canvas.draw_rectangle(0, 0, canvas.width(), canvas.height(), backgroundColor.rgbPtr());
241  }
242  }else{
243  if (redrawCounter >= redrawCounterMax) {
244  redrawCounter = 0;
245  } else {
246  redrawCounter++;
247  return;
248  }
249  }
250 
251  auto latestIter = !hasInvalidated ? std::prev(objects.end(), fromBack) : objects.begin();
252 
253  AffineTransform screen = screentransform();
254  while (latestIter != objects.end()) {
255  SceneObject& object = *latestIter;
256  AffineTransform t(screen.copyConcatenate(object.transform));
257  Color& color = object.color;
258  IDrawableGeometry* geom = object.geom.get();
259  if (!object.text.empty()) {
260  Point trans = t.getTranslation();
261  canvas.draw_text(trans.x, trans.y, object.text.c_str(), object.color.rgbPtr());
262  } else if (object.stamp && object.unownedGeom != nullptr) {
263  object.unownedGeom->draw(screen.copyConcatenate(object.transform), canvas, object.color);
264  } else geom->draw(t, canvas, color);
265  latestIter++;
266  }
267 
268  if(canvas.width() != turtleComposite.width() || canvas.height() || turtleComposite.height()){
269  turtleComposite.assign(canvas);
270  }else{
271  //Let's see if the draw_image is accellerated in some way
272  turtleComposite.draw_image(0,0,canvas);
273  }
274  for (Turtle* turt : turtles) {
275  turt->draw(screen, turtleComposite);
276  }
277 
278  lastTotalObjects = objects.size();
279  display.display(turtleComposite);
280  sleep(delayMS);
281  }

283  void TurtleScreen::initEventThread() {
284  eventThread.reset(new std::thread([&]() {
285  //Mouse button states, between updates.
286  //Keeps track of release/press etc
287  //states for all three mouse buttons for isDown.
288  bool mButtons[3] = {false, false, false};
289  //Same thing for keys here.
290  //(this is a list of keys marked as being in a "down" state)
291  std::list<KeyboardKey> mKeys;
292 
293  while (!display.is_closed() && !killEventThread) {
294  //Updates all input.
295  if(!display.is_event()){
296  std::this_thread::yield();
297  continue;
298  }
299 
300  cacheMutex.lock();
301 
302  AffineTransform mouseOffset = screentransform();
303  Point mousePos = {int((display.mouse_x() - mouseOffset.getTranslateX()) * mouseOffset.getScaleX()),
304  int((display.mouse_y() - mouseOffset.getTranslateY()) * mouseOffset.getScaleY())};
305 
306  //Update mouse button input.
307  const unsigned int button = display.button();
308  bool buttons[3] = {
309  button & 1, //left
310  button & 2, //right
311  button & 4 //middle
312  };
313 
314  for (int i = 0; i < 3; i++) {
315  if (!(!mButtons[i] && buttons[i]))//is this button state "down"?
316  continue; //if not, skip its processing loop.
317 
318  for (MouseFunc& func : mouseBindings[i]) {
319  //append to the event cache.
320  InputEvent e;
321  e.type = false;
322  //perform an inverse translation, then apply scaling.
323  e.mX = mousePos.x;
324  e.mY = mousePos.y;
325  e.cbPointer = reinterpret_cast<void*> (&func);
326  cachedEvents.push_back(e);
327  }
328  }
329 
330  const auto& keys = getKeys();
331 
332  //iterate through every key to determine its state,
333  //then call the appropriate callbacks.
334  for(const auto& keyPair : keys){
335  KeyboardKey key = keyPair.second;
336  const bool lastDown = std::find(mKeys.begin(), mKeys.end(), key) != mKeys.end();
337  const bool curDown = display.is_key((unsigned int) key);
338 
339  int state = -1;
340  if (!lastDown && curDown) {
341  //Key down.
342  state = 0;
343  mKeys.push_back(key);
344  } else if (lastDown && !curDown) {
345  //Key up.
346  state = 1;
347  mKeys.remove(key);
348  }else continue; //skip on case where it was down and is down
349 
350  try {
351  //will throw if no bindings available for key,
352  //and that's perfectly fine, so we just silently catch
353  auto& bindingList = keyBindings[state][key];
354  for (auto& cb : bindingList) {
355  cb();
356  }
357  } catch (...) {}
358  }
359 
360  mButtons[0] = buttons[0];
361  mButtons[1] = buttons[1];
362  mButtons[2] = buttons[2];
363 
364  cacheMutex.unlock();
365  }
366  }));
367  }

551  bool Turtle::undo() {
552  const unsigned long int totalBefore = state.objectsBefore;
553 
554  if(stateStack.size() >= 2)
555  travelBack();//Travel back if stack size >= 2
556 
557  if(!popState()){
558  return false;
559  }
560 
561  auto begin = std::prev(objects.end(), (totalBefore - state.objectsBefore));
562  auto iter = begin;
563 
564  while(iter != objects.end()){
565  screen->getScene().erase(*iter);
566  iter++;
567  }
568 
569  objects.erase(begin, objects.end());
570 
571  //Will invalidate.
572  updateParent(true, false);
573  return true;
574  }

596  void Turtle::travelTo(const AffineTransform& dest) {
597  traveling = true;
598 
599  AffineTransform begin;
600  begin.assign(transform);
601 
602  if (screen != nullptr ? !screen->getIsClosed() : false) {//no point in animating with no screen
603  auto& scene = this->screen->getScene();
604  const float duration = getAnimMS();
605  const unsigned long startTime = epochTime();
606  const unsigned long endTime = duration + startTime;
607 
608  AffineTransform start;
609  start.assign(transform);
610 
611  float progress = duration == 0 ? 1 : 0;
612  while (progress < 1.0f) {
613  //We use the time between animation frames to smooth out
614  //our animations, making them take the same amount of time
615  //regardless of how it's performance.
616  unsigned long curTime = epochTime();
617 
618  transform.assign(start.lerp(dest, progress));
619  travelPoints[0] = begin.getTranslation();
620  travelPoints[1] = transform.getTranslation();
621  screen->redraw();
622 
623  progress = ((curTime - startTime) / duration);
624  std::this_thread::yield();
625  }
626  }
627 
628  //Contents of PushCurrent moved here because every function
629  //that called this one called PushCurrent immediately after it.
630  if(state.tracing){
631  pushTraceLine(begin.getTranslation(), dest.getTranslation());
632  }
633  if (state.filling)
634  fillAccum.points.push_back(transform.getTranslation());
635 
636  transform.assign(begin);
637  pushState();
638  transform.assign(dest);
639 
640  traveling = false;
641  updateParent(false, false);
642  }

644  void Turtle::travelBack(){
645  const AffineTransform a = state.transform;
646  const AffineTransform b = std::prev(stateStack.end(), 2)->transform;
647 
648  if(a == b)
649  return;//no interpolation or animation needed.
650 
651  traveling = true;
652 
653  if (screen != nullptr ? !screen->getIsClosed() : false) {//no point in animating with no screen
654  auto& scene = this->screen->getScene();
655  const float duration = getAnimMS();
656  const unsigned long startTime = epochTime();
657  const unsigned long endTime = duration + startTime;
658 
659  AffineTransform start;
660  start.assign(a);
661 
662  float progress = duration == 0 ? 1 : 0;
663  while (progress < 1.0f) {
664  //We use the time between animation frames to smooth out
665  //our animations, making them take the same amount of time
666  //regardless of how it's performance.
667  unsigned long curTime = epochTime();
668 
669  transform.assign(start.lerp(b, progress));
670  travelPoints[0] = b.getTranslation();
671  travelPoints[1] = transform.getTranslation();
672  screen->redraw();
673 
674  progress = ((curTime - startTime) / duration);
675  std::this_thread::yield();
676  }
677  }
678 
679  state.transform.assign(b);
680 
681  traveling = false;
682  updateParent(false, false);
683  }
